def skip_to_before_line(file, prefix, offset):
    old_position = file.tell()

    while True:
        file.seek(offset, os.SEEK_CUR)

        next(file, None)
        line = next(file, None)
        # print("jumped to", (line or b'<eof>').decode().rstrip())

        if not line or line > prefix:
            file.seek(old_position)
            break

        old_position = file.tell()
